# ─────────────────────────────────────────────────────────
# _config.yml (my-site/_config.yml)
# ─────────────────────────────────────────────────────────
# 탭 대신 스페이스. 들여쓰기 주의!
--- CONFIG START ---
title: "bong bong — portfolio"
description: "Personal blog & projects"
# GitHub Pages에서 기본 테마로 minima 사용
# (원하면 나중에 다른 테마로 교체 가능)
theme: minima

# 개인 사이트( username.github.io )라면 url만 내 계정으로, baseurl은 빈 문자열
url: "https://<your-username>.github.io"  # ← 본인 GitHub 사용자명으로 변경
baseurl: ""  # 개인 사이트면 "".
# 프로젝트 사이트(임의의 저장소)라면 baseurl: "/<repo-name>"

plugins:
  - jekyll-redirect-from

# 프로젝트를 별도 컬렉션으로 관리
collections:
  projects:
    output: true
    permalink: /projects/:name/
--- CONFIG END ---
